Quick Start

Create main.lua:

function RL.init()
  RL.SetWindowTitle("Hello")
end

function RL.update(dt)
end

function RL.draw()
  RL.ClearBackground(RL.RAYWHITE)
  RL.DrawText("Hello!",190,200,20,RL.BLACK)
end

Run: ReiLua.exe
